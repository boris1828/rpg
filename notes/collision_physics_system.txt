colliding shapes:

  AABB
  AABBRamp
  OBB
  Ray
  Segment

types of physics-colliding objects:

  real_static  : AABB    : NoPhysics : pieces of the world that do not change positions
  static_stair : AARamp  : NoPhysics : world stair pieces that do not change positions
  mov_static   : AABB    : NoPhysics : pieces of the world that change positions according to some path
  dynamic      : AABB    : Dynaimic  : entity that decides their movement
  projectile   : Segment : Particle  : entity shot by onether entity
  kynetic      : OBB     : RigidBody : entity passive to any other collision

collision grain:

  projectile_collidable
  dynamic_collidable

collisions description:

  real_static  <-> real_static  : nothing happens
  real_static  <-> mov_static   : nothing happens
  real_static  <-> static_stair : nothing happens
  mov_static   <-> mov_static   : nothing happens
  mov_static   <-> static_stair : nothing happens
  static_stair <-> static_stair : nothing happens

  dynamic <-> real_static       : dynamic entity is moved to resolve collision
  dynamic <-> mov_static        : dynamic entity is moved to resolve collision
  dynamic <-> static_stair      : dynamic entity is moved only upwards to resolve collision
  dynamic <-> dynamic           : both entity are moved halfway to resolve collision

  projectile <-> real_static    : destroy projectile entity and generate idle projectile with last known position and orientation
  projectile <-> static_stair   : destroy projectile entity and generate idle projectile with last known position and orientation
  projectile <-> mov_static     : destroy projectile entity and generate idle projectile with attachment to the hit movable static entity
  projectile <-> dynamic        : destroy projectile entity and generate idle projectile with attachment to the hit dynamic entity
  projectile <-> projectile     : nothing happens

  kinetic <-> real_static       : kinetic entity is moved to resolve collision
  kinetic <-> static_stair      : kinetic entity is moved to resolve collision
  kinetic <-> mov_static        : kinetic entity is moved to resolve collision
  kinetic <-> dynamic           : kinetic entity is moved to resolve collision
  kinetic <-> projectile        : kinetic entity is given impact from projectile collision, destroy projectile entity and generate idle projectile with attachment to the kinetic component
  kinetic <-> kinetic           : both entity are moved to resolve collision

AABB tree:

  StaticAABBTree   : 
    contains the AABB approximations of all the static objects, it is build once at the begininng of the game; 
    the static objects are not entity in the ECS;
  DynamicAABBTree : 
    contains all the AABB approximations of the objects that are entity in the ECS;

// AABB TREE

#define AABB_TREE_MAX_OBJECTS 1024
#define nullAABBTreeIndex     -1 

Real3 min(Real3 v1, Real3 v2) {
  return Real3(min(v1.x, v2.x), min(v1.y, v2.y), min(v1.z, v2.z));
}

Real3 max(Real3 v1, Real3 v2) {
  return Real3(max(v1.x, v2.x), max(v1.y, v2.y), max(v1.z, v2.z));
}

AABB AABBUnion(AABB b1, AABB b2) {
  AABB u;
  u.minv = min(b1.minv, b2.minv);
  u.maxv = max(b1.maxv, b2.maxv);
  return u;
}

Real AABBSurfaceArea(AABB A) {
  Real3 d = A.maxv – A.minv;
  return 2.0f * (d.x * d.y + d.y * d.z + d.z * d.x);
}

struct Node {
  AABB box;
  int parent_idx;
  int child1;
  int child2;
  bool is_leaf;
  Real inherited_cost;
};

struct AABBTree {
  Nodes nodes[AABB_TREE_MAX_OBJECTS];
  int node_count;
  int root_idx;
};

AABBTree createAABBTree() {
  AABBTree tree;
  tree.node_count = 0;
  tree.root_idx   = nullAABBTreeIndex;
  return tree;
}

int allocateLeafNode(AABBTree &tree, AABB box) {
  Node node;
  node.box        = box;
  node.parent_idx = nullAABBTreeIndex;
  node.child1     = nullAABBTreeIndex;
  node.child2     = nullAABBTreeIndex;
  node.is_leaf    = true;

  tree.nodes[tree.node_count] = node;
  return node_count;
}

int allocateInternalNode(AABBTree &tree) {
  Node node;
  node.parent_idx = nullAABBTreeIndex;
  node.child1     = nullAABBTreeIndex;
  node.child2     = nullAABBTreeIndex;
  node.is_leaf    = false;

  tree.nodes[tree.node_count] = node;
  return node_count;
}

int pickBestSibling(AABBTree &tree, int leaf_idx) {

  Node leaf_node = tree.nodes[leaf_idx];

  int best_idx   = tree.root_idx;
  Real best_cost = AABBSurfaceArea(AABBUnion(tree.nodes[tree.root_idx].box, leaf_node.box));
  Stack stack;
  stack.push(tree.root_idx);

  while (!stack.empty()) {

    int node_idx = stack.pop();
    Node &node   = tree.nodes[node_idx];

    Real inherited_cost = 0.0;

    if (node.parent_idx != nullAABBTreeIndex)
      inherited_cost = tree.nodes[node.parent_idx].inherited_cost;

    AABB union_aabb = AABBUnion(node.box, leaf_node.box);
    Real union_sa   = AABBSurfaceArea(union_aabb);
    Real node_sa    = AABBSurfaceArea(node.box);
    node.inherited_cost = (union_sa - node_sa) + inherited_cost;

    Real cost = union_sa + inherited_cost;

    if (cost < best_cost) {
      best_cost = cost;
      best_idx = node_idx;
    }

    // consider adding children to the queue

    if (node.is_leaf) continue;

    Real leaf_node_sa   = AABBSurfaceArea(leaf_node.box);
    Real low_bound_cost = leaf_node_sa + node.inherited_cost;

    if (low_bound_cost < best_cost) {
      stack.push(node.child1);
      stack.push(node.child2);
    }
  }

  return best_idx;
}

int insertLeaf(AABBTree &tree, AABB box) {

  int leaf_idx = allocateLeafNode(tree, box);

  if (tree.node_count == 0) {
    tree.root_idx = leaf_idx;
    tree.node_count++;
    return leaf_idx;
  }

  // Stage 1: find the best sibling for the new leaf

  int sibling = pickBestSibling(tree, leaf_idx);

  // Stage 2: create a new parent

  int old_parent = tree.nodes[sibling].parent_idx;
  int new_parent = AllocateInternalNode(tree);
  tree.nodes[new_parent].parent_idx = old_parent;
  tree.nodes[new_parent].box = AABBUnion(box, tree.nodes[sibling].box);

  if (old_parent != nullAABBTreeIndex)
  {
    // The sibling was not the root
    if (tree.nodes[old_parent].child1 == sibling)
      tree.nodes[old_parent].child1 = new_parent;
    else
      tree.nodes[old_parent].child2 = new_parent;

    tree.nodes[new_parent].child1   = sibling;
    tree.nodes[new_parent].child2   = leaf_idx;
    tree.nodes[sibling].parent_idx  = new_parent;
    tree.nodes[leaf_idx].parent_idx = new_parent;
  }
  else
  {
    // The sibling was the root
    tree.nodes[new_parent].child1   = sibling;
    tree.nodes[new_parent].child2   = leaf_idx;
    tree.nodes[sibling].parent_idx  = new_parent;
    tree.nodes[leaf_idx].parent_idx = new_parent;
    tree.rootIndex = new_parent;
  }

  tree.node_count += 2;

  // Stage 3: walk back up the tree refitting AABBs

  int index = tree.nodes[leaf_idx].parent_idx;
  while (index != nullAABBTreeIndex)
  {
    int child1 = tree.nodes[index].child1;
    int child2 = tree.nodes[index].child2;
    tree.nodes[index].box = AABBUnion(tree.nodes[child1].box, tree.nodes[child2].box);
    index = tree.nodes[index].parent_idx;
  }

  return leaf_idx;
}
