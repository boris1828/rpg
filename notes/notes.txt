enum class TriggerMode : uint8_t
{
    OnAreaFirstEnter, // First ever frame of overlap
    OnAreaEnter,      // First frame of overlap, can be exited and re-entered
    OnAreaExit,       // Frame overlap ceases
    OnAreaStay,       // Every frame of overlap (Continuous)
    OnAreaTick,       // Periodically while overlapping (Per-Entity Timer)
    OnGlobalTick,     // Periodically while overlapping (Global timer)
    OnClick,          // Clicked while hovered/pointed
    OnHover           // While mouse is over (UI feedback)
};


OnAreaFirstEnter: keep a set of all entities that have overlapped
  on reset              : do nothing
  on overlap_collection : collection insert overlapping entity
  on resolution         : apply interaction on new overlapping entities

OnAreaEnter: keep a set of all entities that are overlapping and the frame before entities that were overlapping
  on reset              : put last frame set in old set (old_set = new_set)
  on overlap_collection : insert overlapping entity in new_set
  on resolution         : apply interaction on entities in new_set - old_set

OnAreaExit: keep a set of all entities that are overlapping and the frame before entities that were overlapping
  on reset              : put last frame set in old set (old_set = new_set)
  on overlap_collection : insert overlapping entity in new_set
  on resolution         : apply interaction on entities in old_set - new_set

OnAreaStay: keep a set of all entities that are overlapping
  on reset              : clear set
  on overlap_collection : insert overlapping entity in set
  on resolution         : apply interaction on entities in set

{

    // Filters

    bool affects_actor;

    bool player_was_inside  = false;
    bool player_has_entered = false;

    // State for anonymous interactables
    uint32_t overlap_count          = 0;
    uint32_t previous_overlap_count = 0;    

    // UI State
    bool hovered = false;

    // TODO: runtime/static collision shape: maybe make a separate struct that has them both an rewrite logic on that (cleaner)
    // BVH/Shape Data
    CollisionShape rest_shape;
    CollisionShape runtime_shape;

    InteractableInfo(
        TriggerMode     trigger_mode, 
        InteractionType interaction_type, 
        bool only_player,
        bool clickable,
        bool affects_actor,
        CollisionShape rest_shape, 
        CollisionShape runtime_shape) 
    : 
        trigger_mode(trigger_mode),
        interaction_type(interaction_type),
        only_player(only_player),
        clickable(clickable),
        affects_actor(affects_actor), 
        rest_shape(rest_shape),
        interaction_runtime_shape(runtime_shape)
    {}

    void reset() 
    { 
        player_has_entered = player_has_entered || player_is_inside;
        player_was_inside  = player_is_inside;
